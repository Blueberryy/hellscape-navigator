class m8f_hn_QuestPointer : Actor
{
  Default
  {
    +NOBLOCKMAP;
    +NOGRAVITY;
    +DONTSPLASH;
    +NOTONAUTOMAP;
    +DONTTHRUST;
  }

  override void BeginPlay()
  {
    let eventHandler = m8f_hn_EventHandler(StaticEventHandler.Find("m8f_hn_EventHandler"));
    _id = eventHandler.AddPointer(pos.x, pos.y, m8f_hn_Pointer.POINTER_GOLD);
  }

  override void Die(Actor source, Actor inflictor, int dmgflags)
  {
    let eventHandler = m8f_hn_EventHandler(StaticEventHandler.Find("m8f_hn_EventHandler"));
    eventHandler.RemovePointer(_id);
  }

  int _id;

} // class m8f_hn_QuestPointer
