class m8f_hn_FadingMarker : MapMarker
{
  Default
    {
      RenderStyle "Translucent";
      XScale 0.04;
      YScale 0.04;
    }

  float fadeStep;

  m8f_hn_FadingMarker init(int lifetime)
  {
    fadeStep = 1.0 / lifetime;
    return self;
  }

  override void Tick()
  {
    alpha -= fadeStep;
    if (alpha <= 0.0) { Destroy(); }
  }

  States
    {
    Spawn:
      HNF1 A -1;
      Stop;
    }
}
