class m8f_hn_Pointer
{

  enum PointerTypes
  {
    POINTER_WHITE,
    POINTER_BLUE,
    POINTER_GOLD,
    POINTER_GREEN,
    POINTER_ICE,
    POINTER_RED,
    POINTER_SKIP
  };

  m8f_hn_Pointer init(double x, double y, int type)
  {
    _position.x = x;
    _position.y = y;
    _type       = type;

    return self;
  }

  double x()
  {
    return _position.x;
  }

  double y()
  {
    return _position.y;
  }

  int type()
  {
    return _type;
  }

  vector2 _position;
  int     _type;

} // m8f_hn_Pointer
